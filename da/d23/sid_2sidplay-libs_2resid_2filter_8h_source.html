28 // ----------------------------------------------------------------------------
29 // The SID filter is modeled with a two-integrator-loop biquadratic filter,
30 // which has been confirmed by Bob Yannes to be the actual circuit used in
31 // the SID chip.
32 //
33 // Measurements show that excellent emulation of the SID filter is achieved,
34 // except when high resonance is combined with high sustain levels.
35 // In this case the SID op-amps are performing less than ideally and are
36 // causing some peculiar behavior of the SID filter. This however seems to
37 // have more effect on the overall amplitude than on the color of the sound.
38 //
39 // The theory for the filter circuit can be found in "Microelectric Circuits"
40 // by Adel S. Sedra and Kenneth C. Smith.
41 // The circuit is modeled based on the explanation found there except that
42 // an additional inverter is used in the feedback from the bandpass output,
43 // allowing the summer op-amp to operate in single-ended mode. This yields
44 // inverted filter outputs with levels independent of Q, which corresponds with
45 // the results obtained from a real SID.
46 //
47 // We have been able to model the summer and the two integrators of the circuit
48 // to form components of an IIR filter.
49 // Vhp is the output of the summer, Vbp is the output of the first integrator,
50 // and Vlp is the output of the second integrator in the filter circuit.
51 //
52 // According to Bob Yannes, the active stages of the SID filter are not really
53 // op-amps. Rather, simple NMOS inverters are used. By biasing an inverter
54 // into its region of quasi-linear operation using a feedback resistor from
55 // input to output, a MOS inverter can be made to act like an op-amp for
56 // small signals centered around the switching threshold.
57 //
58 // Qualified guesses at SID filter schematics are depicted below.
59 //
60 // SID filter
61 // ----------
62 //
63 // -----------------------------------------------
64 // | |
65 // | ---Rq-- |
66 // | | | |
67 // | ------------<A]-----R1--------- |
68 // | | | |
69 // | | ---C---| ---C---|
70 // | | | | | |
71 // | --R1-- ---R1-- |---Rs--| |---Rs--|
72 // | | | | | | | |
73 // ----R1--|-----[A>--|--R-----[A>--|--R-----[A>--|
74 // | | | |
75 // vi -----R1-- | | |
76 //
77 // vhp vbp vlp
78 //
79 //
80 // vi - input voltage
81 // vhp - highpass output
82 // vbp - bandpass output
83 // vlp - lowpass output
84 // [A> - op-amp
85 // R1 - summer resistor
86 // Rq - resistor array controlling resonance (4 resistors)
87 // R - NMOS FET voltage controlled resistor controlling cutoff frequency
88 // Rs - shunt resitor
89 // C - capacitor
90 //
91 //
92 //
93 // SID integrator
94 // --------------
95 //
96 // V+
97 //
98 // |
99 // |
100 // -----|
101 // | |
102 // | ||--
103 // -||
104 // ---C--- ||->
105 // | | |
106 // |---Rs-----------|---- vo
107 // | |
108 // | ||--
109 // vi ---- -----|------------||
110 // | ^ | ||->
111 // |___| | |
112 // ----- | |
113 // | | |
114 // |---R2-- |
115 // |
116 // R1 V-
117 // |
118 // |
119 //
120 // Vw
121 //
122 // ----------------------------------------------------------------------------

215 // ----------------------------------------------------------------------------
216 // SID clocking - 1 cycle.
217 // ----------------------------------------------------------------------------
218 RESID_INLINE
219 void Filter::clock(sound_sample voice1,
220  sound_sample voice2,
221  sound_sample voice3)
222 {
223  // Add separate DC offset to each voice.
224  // Scale each voice down from 20 to 13 bits.
225  voice1 = voice1 >> 7;
226  voice2 = voice2 >> 7;
227 
228  // NB! Voice 3 is not silenced by voice3off if it is routed through
229  // the filter.
230  if (voice3off && !(filt3_filt2_filt1 & 0x04)) {
231  voice3 = 0;
232  }
233  else {
234  voice3 = voice3 >> 7;
235  }
236 
237  // This is handy for testing.
238  if (!enabled) {
239  Vnf = voice1 + voice2 + voice3;
240  Vhp = Vbp = Vlp = 0;
241  return;
242  }
243 
244  // Route voices into or around filter.
245  // The code below is expanded to a switch for faster execution.
246  // (filt1 ? Vi : Vnf) += voice1;
247  // (filt2 ? Vi : Vnf) += voice2;
248  // (filt3 ? Vi : Vnf) += voice3;
249 
250  sound_sample Vi;
251 
252  switch (filt3_filt2_filt1) {
253  default:
254  case 0x0:
255  Vi = 0;
256  Vnf = voice1 + voice2 + voice3;
257  break;
258  case 0x1:
259  Vi = voice1;
260  Vnf = voice2 + voice3;
261  break;
262  case 0x2:
263  Vi = voice2;
264  Vnf = voice1 + voice3;
265  break;
266  case 0x3:
267  Vi = voice1 + voice2;
268  Vnf = voice3;
269  break;
270  case 0x4:
271  Vi = voice3;
272  Vnf = voice1 + voice2;
273  break;
274  case 0x5:
275  Vi = voice1 + voice3;
276  Vnf = voice2;
277  break;
278  case 0x6:
279  Vi = voice2 + voice3;
280  Vnf = voice1;
281  break;
282  case 0x7:
283  Vi = voice1 + voice2 + voice3;
284  Vnf = 0;
285  break;
286  }
287 
288  // delta_t = 1 is converted to seconds given a 1MHz clock by dividing
289  // with 1 000 000.
290 
291  // Calculate filter outputs.
292  // Vhp = Vbp/Q - Vlp - Vi;
293  // dVbp = -w0*Vhp*dt;
294  // dVlp = -w0*Vbp*dt;
295 
296  sound_sample dVbp = (w0*Vhp >> 20);
297  sound_sample dVlp = (w0*Vbp >> 20);
298  Vbp -= dVbp;
299  Vlp -= dVlp;
300  Vhp = (Vbp*_1024_div_Q >> 10) - Vlp - Vi;
301 }

303 // ----------------------------------------------------------------------------
304 // SID clocking - delta_t cycles.
305 // ----------------------------------------------------------------------------
306 RESID_INLINE
307 void Filter::clock(cycle_count delta_t,
308  sound_sample voice1,
309  sound_sample voice2,
310  sound_sample voice3)
311 {
312  // Add separate DC offset to each voice.
313  // Scale each voice down from 20 to 13 bits.
314  voice1 = voice1 >> 7;
315  voice2 = voice2 >> 7;
316 
317  // NB! Voice 3 is not silenced by voice3off if it is routed through
318  // the filter.
319  if (voice3off && !(filt3_filt2_filt1 & 0x04)) {
320  voice3 = 0;
321  }
322  else {
323  voice3 = voice3 >> 7;
324  }
325 
326  // Enable filter on/off.
327  // This is not really part of SID, but is useful for testing.
328  // On slow CPUs it may be necessary to bypass the filter to lower the CPU
329  // load.
330  if (!enabled) {
331  Vnf = voice1 + voice2 + voice3;
332  Vhp = Vbp = Vlp = 0;
333  return;
334  }
335 
336  // Route voices into or around filter.
337  // The code below is expanded to a switch for faster execution.
338  // (filt1 ? Vi : Vnf) += voice1;
339  // (filt2 ? Vi : Vnf) += voice2;
340  // (filt3 ? Vi : Vnf) += voice3;
341 
342  sound_sample Vi;
343 
344  switch (filt3_filt2_filt1) {
345  default:
346  case 0x0:
347  Vi = 0;
348  Vnf = voice1 + voice2 + voice3;
349  break;
350  case 0x1:
351  Vi = voice1;
352  Vnf = voice2 + voice3;
353  break;
354  case 0x2:
355  Vi = voice2;
356  Vnf = voice1 + voice3;
357  break;
358  case 0x3:
359  Vi = voice1 + voice2;
360  Vnf = voice3;
361  break;
362  case 0x4:
363  Vi = voice3;
364  Vnf = voice1 + voice2;
365  break;
366  case 0x5:
367  Vi = voice1 + voice3;
368  Vnf = voice2;
369  break;
370  case 0x6:
371  Vi = voice2 + voice3;
372  Vnf = voice1;
373  break;
374  case 0x7:
375  Vi = voice1 + voice2 + voice3;
376  Vnf = 0;
377  break;
378  }
379 
380  // Maximum delta cycles for the filter to work satisfactorily under current
381  // cutoff frequency and resonance constraints is approximately 8.
382  cycle_count delta_t_flt = 8;
383 
384  // Limit f0 to 4kHz to keep filter stable.
385  const double pi = 3.1415926535897932385;
386  const sound_sample w0_max = static_cast<sound_sample>(2*pi*4000*1.048576);
387  sound_sample w0_ceil = w0 <= w0_max ? w0 : w0_max;
388 
389  while (delta_t) {
390  if (delta_t < delta_t_flt) {
391  delta_t_flt = delta_t;
392  }
393 
394  // delta_t is converted to seconds given a 1MHz clock by dividing
395  // with 1 000 000. This is done in two operations to avoid integer
396  // multiplication overflow.
397 
398  // Calculate filter outputs.
399  // Vhp = Vbp/Q - Vlp - Vi;
400  // dVbp = -w0*Vhp*dt;
401  // dVlp = -w0*Vbp*dt;
402  sound_sample w0_delta_t = w0_ceil*delta_t_flt >> 6;
403 
404  sound_sample dVbp = (w0_delta_t*Vhp >> 14);
405  sound_sample dVlp = (w0_delta_t*Vbp >> 14);
406  Vbp -= dVbp;
407  Vlp -= dVlp;
408  Vhp = (Vbp*_1024_div_Q >> 10) - Vlp - Vi;
409 
410  delta_t -= delta_t_flt;
411  }
412 }

416 // SID audio output (20 bits).
417 // ----------------------------------------------------------------------------
418 RESID_INLINE
419 sound_sample Filter::output()
420 {
421  // This is handy for testing.
422  if (!enabled) {
423  return (Vnf + mixer_DC)*static_cast<sound_sample>(vol);
424  }
425 
426  // Mix highpass, bandpass, and lowpass outputs. The sum is not
427  // weighted, this can be confirmed by sampling sound output for
428  // e.g. bandpass, lowpass, and bandpass+lowpass from a SID chip.
429 
430  // The code below is expanded to a switch for faster execution.
431  // if (hp) Vf += Vhp;
432  // if (bp) Vf += Vbp;
433  // if (lp) Vf += Vlp;
434 
435  sound_sample Vf;
436 
437  switch (hp_bp_lp) {
438  default:
439  case 0x0:
440  Vf = 0;
441  break;
442  case 0x1:
443  Vf = Vlp;
444  break;
445  case 0x2:
446  Vf = Vbp;
447  break;
448  case 0x3:
449  Vf = Vlp + Vbp;
450  break;
451  case 0x4:
452  Vf = Vhp;
453  break;
454  case 0x5:
455  Vf = Vlp + Vhp;
456  break;
457  case 0x6:
458  Vf = Vbp + Vhp;
459  break;
460  case 0x7:
461  Vf = Vlp + Vbp + Vhp;
462  break;
463  }
464 
465  // Sum non-filtered and filtered output.
466  // Multiply the sum with volume.
467  return (Vnf + Vf + mixer_DC)*static_cast<sound_sample>(vol);
468 }
